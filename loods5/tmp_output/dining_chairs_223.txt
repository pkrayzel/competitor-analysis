        function updateStock() {
            var store = jQuery('#store-stock option:selected').val();
            var vars = jQuery('.product--selectbox--variant.active').attr('id');
            var comboid = vars.split('combination-')[1];
            var decodedString = atob(jQuery('#jsstk').val());
            var obj = jQuery.parseJSON(decodedString);

            if (store) {
                jQuery('.store-result').show();
                if (obj[comboid][store] > 0) {
                    if (obj[comboid][store] >= 10) {
                        jQuery('.store-result .store-result-stock').addClass('in-stock').removeClass('out-of-stock').text('Snel leverbaar');
                    } else {
                        jQuery('.store-result .store-result-stock').addClass('in-stock').removeClass('out-of-stock').text('Nog maar '+ obj[comboid][store] + ' beschikbaar');
                    }
                } else {
                    jQuery('.store-result .store-result-stock').addClass('out-of-stock').removeClass('in-stock').text('Niet op voorraad');
                }
            } else {
                jQuery('.store-result').hide();
            }
        }

        }
